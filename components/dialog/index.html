基于Dialog的MessageBox -- JS调用
打开消息 自定义 

        <a yui-button onclick="openMessage()">打开消息</a>

        <a yui-button type="primary" onclick="openCustomMessage()">自定义</a>

        <script>

            function openMessage() {
                let name = yuiMessageBox('提示', '这是一条消息', {
                    beforeClose: function (name, callback) {
                        yuiMessage(`关闭前的回调`, {delay: 1000, type: "success"});
                        callback()
                    },
                });
                console.log(`生成name='${name}'的MessageBox`);
            }

            function openCustomMessage() {
                let name = yuiMessageBox('提示', '有一条新的消息', {
                    clickMaskNotClosed: true,
                    showLeftBtn: true,
                    leftBtnText: '直接删除',
                    leftBtnMethod: function (name, callback) {
                        yuiMessage(`直接删除`, {type: "danger", delay: 1000});
                        callback();
                    },
                    leftBtnAttr: {type: "danger", size: "mini"},
                    rightBtnText: "查 看",
                    rightBtnAttr: {type: "success", size: "mini"},
                    rightBtnMethod: function (name, callback) {
                        yuiMessage(`查看`, {type: "success", delay: 1000});
                        callback();
                    },
                });
                console.log(`生成name='${name}'的MessageBox`);
            }

        </script>

事件
方法 说明 参数 返回
yuiDialogClosed(name) 关闭对话框的function name：想要关闭的对话框名称 --
yuiMessageBox(title,content,options) 打开MessageBox title：标题
content：内容
options：配置，具体见下表 		生成的MessageBox名称
yuiMessageBox中的options
参数 说明 类型 可选值 / 回调参数 默认值 / 参数形式
showLeftBtn 是否显示左边按钮 boolean true / false false
leftBtnText 左边按钮文字 string -- 取 消
leftBtnAttr 左边按钮样式 object 参考yui-button属性 {size:"small"}
leftBtnMethod 左边按钮点击事件，可以使用 callback() 或者 yuiDialogClosed(name) 进行关闭 function name:MessageBox名称
callback:关闭MessageBox的方法
 (name,callback)
showRightBtn 是否显示右边按钮 boolean true / false true
rightBtnText 右边按钮文字 string -- 确 定
rightBtnAttr 右边按钮样式 object 参考yui-button属性 {size:"small",type:"primary"}
rightBtnMethod 右边按钮点击事件，可以使用 callback() 或者 yuiDialogClosed(name) 进行关闭 function name:MessageBox名称
callback:关闭MessageBox的方法
 (name,callback)
showCloseIcon 是否显示右上角关闭图标 boolean true / false true
clickMaskNotClosed 点击遮罩不关闭MessageBox boolean true / false false
beforeClose 点击关闭图标或点击遮罩时的回调，可以使用 callback() 或者 yuiDialogClosed(name) 或者 return true 进行关闭 function name:MessageBox名称
callback:关闭MessageBox的方法
 (name,callback)
