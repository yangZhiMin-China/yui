Checkbox 多选框
一组备选项中进行多选
多选框组
复选款 A 复选款 B 复选款 C 禁用 选中且禁用
获取状态数据 

        <div yui-checkbox-group id="my-checkbox" change="myCheckboxChange">
            <a value="复选款A" yui-checkbox>复选款 A</a>
            <a value="复选款B" yui-checkbox>复选款 B</a>
            <a value="复选款C" yui-checkbox checked>复选款 C</a>
            <a value="禁用" yui-checkbox disabled>禁用</a>
            <a value="选中且禁用" yui-checkbox checked disabled>选中且禁用</a>
        </div>

        <a onclick="getCheckboxChecked()" type="primary" yui-button plain>获取状态数据</a>

        <script>

            function myCheckboxChange(value, checked) {
                if (value === "复选款A") {
                    yui.message.danger(`拒绝${checked ? '取消选中' : '选中'} ${value}`);
                    return false
                }
                yui.message.info(`${checked ? '取消选中' : '选中'} ${value}`);
            }

            function getCheckboxChecked() {
                const data = yui.checkbox.data("my-checkbox");
                yui.message.success(`选中：${JSON.stringify(data['checked'])}`, {effect: 'dark'});
                yuiMessage(`未选中：${JSON.stringify(data['unchecked'])}`, {effect: 'dark'});
            }

        </script>
