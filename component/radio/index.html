Radio 单选框
在一组备选项中进行单选
单选框组
备选项 A 备选项 B 备选项 C 禁用 禁用并选中
获取选中数据 

        <div yui-radio-group id="my-radio" change="myRadioChange">
            <a value="备选项A" yui-radio>备选项 A</a>
            <a value="备选项B" yui-radio>备选项 B</a>
            <a value="备选项C" yui-radio>备选项 C</a>
            <a value="禁用" yui-radio disabled>禁用</a>
            <a value="禁用并选中" yui-radio checked disabled>禁用并选中</a>
        </div>

        <a onclick="getRadioChecked()" type="primary" yui-button plain>获取选中数据</a>

        <script>

            function myRadioChange(value) {
                if (value === "备选项A") {
                    yui.message.danger(`拒绝选中 ${value}`);
                    return false
                }
                yui.message.success(`选中 ${value}`);
            }

            function getRadioChecked() {
                const value = yui.radio.data('my-radio');
                yui.message.success(`当前选中：${value}`, {effect: 'dark'});
            }

        </script>
