NumRun 数字滚动
数字滚动特效
递增
开始滚动 0 

        <a onclick="startAdd()" yui-button>开始滚动</a>
        <span yui-num-run id="my-add">0</span>

        <script>

            function startAdd() {
                yui.numRun.start('my-add', {
                    fromVal: 0,
                    toVal: 381,
                    callback: (id, dom) => {
                        yui.message.success('数字滚动完成，将会自动重置');
                        setTimeout(() => {
                            dom.innerText = 0;
                        }, 3000)
                    }
                })
            }

        </script>
递减
开始滚动 500 

        <a onclick="startReduce()" yui-button>开始滚动</a>
